    HELIX_RESOURCE {
        guid resourceId PK
        string name "Primary"
        guid systemUserId FK "1:1 with systemuser"
        choice accessLevel "Admin/Manager/PM/Member"
        choice jobRole "Dev/QA/BA/PM/etc"
        decimal capacityHoursPerWeek
        currency hourlyRate "FLS"
        boolean billable
        guid managedBy FK "self-reference"
        boolean active
    }

    HELIX_PROJECT {
        guid projectId PK
        string name "Primary"
        string code "Alt key, unique"
        choice status "Planning/Active/OnHold/Completed"
        choice priority "Critical/High/Medium/Low"
        choice department
        date startDate
        date endDate
        decimal estimatedHours
        currency budget "FLS"
        currency funding "FLS"
        currency spent "FLS rollup"
        calculated fundingGap "FLS"
        guid projectManager FK
        guid teamManager FK
        int healthScore
    }

    HELIX_PHASE {
        guid phaseId PK
        guid projectId FK
        string name
        choice phaseType "Discovery/Design/Build/Test/Deploy"
        date startDate
        date endDate
        int progress
        guid owner FK
    }

    HELIX_DEPENDENCY {
        guid dependencyId PK
        guid fromPhaseId FK
        guid toPhaseId FK
        choice type "FS/SS/FF/SF"
        int lagDays
    }

    HELIX_ALLOCATION {
        guid allocationId PK
        guid projectId FK
        guid resourceId FK
        date weekStartDate "Alt key"
        decimal hoursPerWeek
        string roleOnProject
        boolean committed
    }

    HELIX_SCENARIO {
        guid scenarioId PK
        string name "Primary"
        guid projectId FK
        choice status
        date baselineEndDate
        date proposedEndDate
        int timelineDeltaDays
        currency baselineBudget "FLS"
        currency estimatedCostDelta "FLS"
        decimal efficiencyCoefficient
        guid owner FK
    }

    HELIX_SCENARIO_ALLOCATION {
        guid scenarioAllocationId PK
        guid scenarioId FK
        guid resourceId FK
        choice action "Add/Remove/Modify"
        date weekStartDate
        decimal baselineHours
        decimal proposedHours
        calculated hoursDelta
    }

    HELIX_TIME_ENTRY {
        guid timeEntryId PK
        guid projectId FK
        guid resourceId FK
        date entryDate
        decimal hours
        string description
        boolean approved
    }
ⓘ
Reading the diagram: The ||--o{ notation means "one to many" (one project has many phases). The ||--o| means "one to one or zero" (one systemuser has at most one helix_resource record). Field rows show type name "annotation". FLS-marked fields enforce Admin-only read via Dataverse Field Security Profile.
